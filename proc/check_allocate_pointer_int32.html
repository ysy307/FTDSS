    subroutine Check_Allocate_Pointer_int32(iptr)
        implicit none
        integer(int32), pointer :: iptr
        
        if (.not. associated(iptr)) then
            allocate(iptr)
        else
            call error_message(955)
        end if
    end subroutine Check_Allocate_Pointer_int32
