    subroutine Calc_Area_31(Geometry)
        implicit none
        type(Geometry2d), intent(inout) :: Geometry
        type(Vector2d)                  :: p1, p2, p3
        integer(int32)                  :: elem

        do elem = 1, Geometry%element
            call set_point(Geometry, elem, p1, p2, p3)
            if (elem < 1 .or. elem > Geometry%element) then
                call error_message(932)
            else
                Geometry%eArea(elem) = abs((p2%x - p1%x) * (p3%y - p1%y) - (p2%y - p1%y) * (p3%x - p1%x)) / 2.d0
            end if
        end do

    end subroutine Calc_Area_31
